For this reason, field "res" in template classes "Exp_elem, Ln_elem, Cos_elem, Sin_elem" contains result multiplied by "static const long long accuracy". Convert it to float and divide by "static const long long accuracy" to get result. Do not enter large numbers in absolute numbers. It could call type overflow. Don`t define "iter_exp, iter_log, iter_sin, iter_cos, accuracy" large numbers. It could call stack overflow. Uncomment line with #define DEBUG to see debug information. }; template <int n, int m> class Casc_exp <1, n, m> { public: enum {res = Exp<iter_exp, n, m>::res }; }; //====================================================================================== // EXP //====================================================================================== template <long long k, long long n, long long m> class Ln_elem { public: enum {res = Ln_elem<k - 1, n, m>::res * ( -1 ) * (n - m) * (k - 1) / m / k}; }; template <long long n, long long m> class Ln_elem <1, n, m> { public: enum {res = n * accuracy / m - accuracy}; }; template <long long k, long long n, long long m> class Ln { public: enum {res = Ln<k - 1, n, m>::res + Ln_elem<k, n, m>::res }; }; template <long long n, long long m> class Ln <1, n, m> { public: enum {res = Ln_elem<1, n, m>::res }; }; //Composition of Ln template <int k, int n, int m> class Casc_ln { public: enum {res = Ln<iter_log, Casc_ln<k - 1, n, m>::res , accuracy>::res}; }; template <long long n, long long m> class Cos_elem <0, n, m> { public: enum {res = accuracy}; }; template <long long k, long long n, long long m> class Cos { public: enum {res = Cos<k - 1, n, m>::res + Cos_elem<k, n, m>::res }; }; template <long long n, long long m> class Cos <0, n, m> { public: enum {res = Cos_elem<0, n, m>::res }; }; //Composition of Cos template <int k, int n, int m> class Casc_cos { public: enum {res = Cos<iter_cos, Casc_cos<k - 1, n, m>::res , accuracy>::res}; }; template <long long n, long long m> class Sin_elem <0, n, m> { public: enum {res = n * accuracy / m}; }; template <long long k, long long n, long long m> class Sin { public: enum {res = Sin<k - 1, n, m>::res + Sin_elem<k, n, m>::res };
